250@Override
251public void nextMove(ParserOutput p, Window window) {
252 parserOutput = p;
253 messages.clear();
254
255 if (p.getCommand() == null) {
256 window.showMessage("Non ho capito cosa devo fare! Prova con un altro comando.");
257 } else {
258 Room cr = getCurrentRoom();
259 notifyObservers(window);
260 boolean move = !cr.equals(getCurrentRoom()) && getCurrentRoom() != null;
261
262 // Controllo per il comando LOOK_AT
263 if (p.getCommand().getType() == CommandType.LOOK_AT) {
264 AdvObject object = p.getObject();
265 if (object != null) {
266 window.showMessage("Osservi " + object.getName() + ": " + object.getDescription());
267 } else {
268 window.showMessage(cr.getLook());
269 if (cr.getName().equalsIgnoreCase("Anticamera")) {
270 if (cr.isMonsterAlive()) {
271 cr.setDynamicLook("Sei nell'anticamera, l'alieno gigante si sta svegliando e ti osserva con occhi minacciosi!\n");
272 }
273 window.showMessage(cr.getDynamicLook());
274 }
275 }
276 } else {
277 if (!messages.isEmpty()) {
278 for (String m : messages) {
279 if (m.length() > 0) {
280 window.showMessage(m);
281 }
282 }
283 }
284
285 if (move) {
286 Room currentRoom = getCurrentRoom();
287 if (currentRoom.getName().equals("Sala motori")) {
288 boolean hasKey = getInventory().stream().anyMatch(obj -> obj.getId() == 5);
289 if (hasKey) {
290 if (isKeyUsed()) {
291 window.showRoomName(currentRoom.getName());
292 window.showRoomDescription(currentRoom.getDescription());
293 } else {
294 window.showMessage("La porta è chiusa ma hai la chiave. Usa la chiave per aprirla.");
295 setCurrentRoom(cr);
296 }
297 } else {
298 window.showMessage("La porta è chiusa e non puoi entrare perché non hai la chiave.");
299 setCurrentRoom(cr); // Torna alla stanza precedente
300 }
301 } else {
302 window.showRoomName(currentRoom.getName());
303 window.showRoomDescription(currentRoom.getDescription());
304 }
305 } else {
306 // Se non c'è stato un movimento, mostra la stanza corrente
307 window.showRoomName(cr.getName());
308 window.showRoomDescription(cr.getDescription());
309 }
310 }
311 }
