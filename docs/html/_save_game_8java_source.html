102 } else if (line.startsWith("CurrentRoom:")) {
103 gameData.put("CurrentRoom", Integer.valueOf(line.substring(12).trim()));
104 } else if (line.startsWith("MonsterAlive:")) {
105 gameData.put("MonsterAlive", Boolean.valueOf(line.substring(13).trim()));
106 } else if (line.startsWith("IsDoorOpen:")) {
107 gameData.put("IsDoorOpen", Boolean.valueOf(line.substring(11).trim()));
108 } else if (line.startsWith("Inventory:")) {
109 String[] inventoryIds = line.substring(10).trim().split(",");
110 List<Integer> inventory = new ArrayList<>();
111 for (String id : inventoryIds) {
112 if (!id.isEmpty()) {
113 inventory.add(Integer.valueOf(id.trim()));
114 }
115 }
116 gameData.put("Inventory", inventory);
117 } else if (line.equals("---")) {
118 break;
119 }
120 }
121 }
122 } catch (IOException e) {
123 }
