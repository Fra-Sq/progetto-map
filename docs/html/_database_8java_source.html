45 public boolean insertNewData() {

49 System.out.print("Inserisci il Nome: ");

54 String look = scanner.nextLine();

56 try {
57 String query = "INSERT INTO descrizioni (ID, NOME, DESCRIZIONE, LOOK) VALUES (?, ?, ?, ?)";
58 try (PreparedStatement ps = conn.prepareStatement(query)) {
59 ps.setString(1, id); // Imposta l'ID
60 ps.setString(2, nome); // Imposta il Nome
61 ps.setString(3, descrizione); // Imposta la Descrizione
62 ps.setString(4, look); // Imposta il Look
