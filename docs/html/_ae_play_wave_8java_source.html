83 FloatControl pan = (FloatControl) auline.getControl(FloatControl.Type.PAN);
84 if (curPosition == Position.RIGHT)
85 pan.setValue(1.0f);
86 else if (curPosition == Position.LEFT)
87 pan.setValue(-1.0f);

101 } finally {
102 auline.drain();
103 auline.close();
