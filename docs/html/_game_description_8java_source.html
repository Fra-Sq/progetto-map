59 .findFirst();
60 room.ifPresent(r -> this.currentRoom = r);
61
62 this.inventory = new ArrayList<>();
63 for (int id : inventoryIds) {
64 Optional<AdvObject> obj = getObjectById(id);
65 obj.ifPresent(o -> {
66 this.inventory.add(o);
67 // Rimuovi l'oggetto dalla stanza in cui si trova
68 for (Room stanze : rooms) {
69 if (stanze.getObjects().contains(o)) {
70 stanze.getObjects().remove(o);
71 }
72 if (o.getId() == 6) {
73 AdvObject safe = stanze.getObject(4);

75 safe.setOpenable(false);
76 }
77 }
78 if (stanze.getId()==8 && monsterAlive==false) {
