63 public ParserOutput parse(String command, List<Command> commands, List<AdvObject> objects, List<AdvObject> inventory) {
64 List<String> tokens = Utils.parseString(command, stopwords);
65 if (!tokens.isEmpty()) {
66 int ic = checkForCommand(tokens.get(0), commands);
67 if (ic > -1) {
68 if (tokens.size() > 1) {
69 int io = checkForObject(tokens.get(1), objects);
70 int ioinv = -1;
71 if (io < 0 && tokens.size() > 2) {
72 io = checkForObject(tokens.get(2), objects);
73 }
74 if (io < 0) {
75 ioinv = checkForObject(tokens.get(1), inventory);
76 if (ioinv < 0 && tokens.size() > 2) {
77 ioinv = checkForObject(tokens.get(2), inventory);
78 }
79 }
80 if (io > -1 && ioinv > -1) {
81 return new ParserOutput(commands.get(ic), objects.get(io), inventory.get(ioinv));
82 } else if (io > -1) {
83 return new ParserOutput(commands.get(ic), objects.get(io), null);
84 } else if (ioinv > -1) {
85 return new ParserOutput(commands.get(ic), null, inventory.get(ioinv));
86 } else {
87 return new ParserOutput(commands.get(ic), null, null);
88 }
